#include <stdio.h>
#include <stdlib.h>


void swap(int *xp, int *yp)
{
    int temp = *xp;
    *xp = *yp;
    *yp = temp;
}

void bubbleSort(int arr[], int e[], int n)
{
   int i, j;
   for (i = 0; i < n-1; i++)
       for (j = 0; j < n-i-1; j++)
           if (arr[j] > arr[j+1])
           {
               swap(&arr[j], &arr[j+1]);
               swap(&e[j], &e[j+1]);
           }
}

int main()
{
    int t, t1, n, i;
    scanf("%d", &t);
    t1 = t;

    while(t--)
    {
        int flag = 0;
        scanf("%d", &n);
        int s[n], e[n], s1[n];
        char result[n];

        for(i=0; i<n; i++)
        {
            scanf("%d %d", &s[i], &e[i]);
            s1[i] = s[i];
        }

        bubbleSort(s, e, n);

        int max_c=0,max_j=0,min_c=0,min_j=0,c=0,j=0;

        for(i=0; i<n; i++)
        {
            if(s[i]>=max_c)
                c=0;
            if(s[i]>=max_j)
                j=0;
            if(c==0)
            {
                c=1;
                min_c=s[i];
                max_c=e[i];
                result[i]='C';
            }
            else if(j==0)
            {
                j=1;
                min_j=s[i];
                max_j=e[i];
                result[i]='J';
            }
            else
            {
                flag=1;
            }
        }

        printf("Case #%d: ", (t1-t));
        if(flag==1)
        {
            printf("IMPOSSIBLE\n");
        }
        else
        {
            for(i=0; i<n; i++)
            {
                int fix = 0;
                if(s1[fix] == s[i])
                {
                    swap(&result[fix], &result[i]);
                    fix++;
                }
            }
            for(i=0; i<n; i++)
                printf("%c", result[i]);
            printf("\n");
        }
    }
}
